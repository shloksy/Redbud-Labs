141 // event handled? (the most frequent case)
142 if (r >= Q_RET_HANDLED) {
143 break; // done scanning the state hierarchy
144 }
145 // event unhandled and passed to the superstate?
146 else if (r == Q_RET_SUPER) {
147 t = t->superstate; // advance to the superstate
148 }
149 // event unhandled and passed to a submachine superstate?

152 }
153 // event unhandled due to a guard?

155
156 QS_BEGIN_PRE_(QS_QEP_UNHANDLED, qs_id)
157 QS_SIG_PRE_(e->sig); // the signal of the event
158 QS_OBJ_PRE_(this); // this state machine object
159 QS_FUN_PRE_(t->stateHandler); // the current state
160 QS_END_PRE_()
161
162 t = t->superstate; // advance to the superstate
163 }

167 }
168 } while (t != nullptr);

170 // any kind of transition taken?
171 if (r >= Q_RET_TRAN) {

174
175 // the transition source state must not be nullptr

177 #endif // Q_SPY

179 do {

190 }

208 #endif // Q_SPY

437//${QEP::QMsm::enterHistory_} ................................................
438QState QMsm::enterHistory_(
439 QMState const * const hist,
440 std::uint_fast8_t const qs_id)
441{
442 QMState const *s = hist;
443 QMState const *ts = m_state.obj; // transition source
444 QMState const *epath[MAX_ENTRY_DEPTH_];
445
446 QS_CRIT_STAT_
447
448 QS_BEGIN_PRE_(QS_QEP_TRAN_HIST, qs_id)
449 QS_OBJ_PRE_(this); // this state machine object
450 QS_FUN_PRE_(ts->stateHandler); // source state handler
451 QS_FUN_PRE_(hist->stateHandler); // target state handler
452 QS_END_PRE_()
453
454 std::int_fast8_t i = 0; // entry path index
455 while (s != ts) {
456 if (s->entryAction != nullptr) {
457 Q_ASSERT_ID(620, i < MAX_ENTRY_DEPTH_);
458 epath[i] = s;
459 ++i;
460 }
461 s = s->superstate;
462 if (s == nullptr) {
463 ts = s; // force exit from the for-loop
464 }
465 }
466
467 // retrace the entry path in reverse (desired) order...
468 while (i > 0) {
469 --i;
470 // run entry action in epath[i]
471 (*epath[i]->entryAction)(this);
472
473 QS_BEGIN_PRE_(QS_QEP_STATE_ENTRY, qs_id)
474 QS_OBJ_PRE_(this);
475 QS_FUN_PRE_(epath[i]->stateHandler); // entered state handler
476 QS_END_PRE_()
477 }
478
479 m_state.obj = hist; // set current state to the transition target
480
481 // initial tran. present?
482 QState r;
483 if (hist->initAction != nullptr) {
484 r = (*hist->initAction)(this); // execute the transition action
485 }
486 else {
487 r = Q_RET_NULL;
488 }
489
490 Q_UNUSED_PAR(qs_id); // when Q_SPY not defined
491 return r;
492}
