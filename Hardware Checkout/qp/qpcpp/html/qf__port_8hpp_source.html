33//! The maximum number of active objects in the application.
34//!
35//! @description
36//! This macro *must* be defined in the QF port and should be in range
37//! of 1U..64U, inclusive. The value of this macro determines the maximum
38//! priority level of an active object in the system. Not all priority
39//! levels must be used, but the maximum priority cannot exceed
40//! #QF_MAX_ACTIVE.
41#define QF_MAX_ACTIVE 32U
42
43//! The maximum number of clock tick rates in the application.
44//!
45//! @description
46//! This macro can be defined in the QF ports and should be in range
47//! of 1U..15U, inclusive. The value of this macro determines the maximum
48//! number of clock tick rates for time events (::QTimeEvt).
49//!
50//! If the macro is not defined, the default value is 1U.
51#define QF_MAX_TICK_RATE 1U

53//! The maximum number of event pools in the application.
54//!
55//! This macro should be defined in the QF port and should be in range
56//! of 1..255, inclusive. The value of this macro determines the maximum
57//! number of event pools in the system. Not all all these event pools must
58//! actually be used by the applications, but the maximum number of pools
59//! cannot exceed #QF_MAX_EPOOL.
60//!
61//! If the macro is not defined, the default value is 3
62#define QF_MAX_EPOOL 3U

72//! The size (in bytes) of the ring-buffer counters used in the native
73//! QF event queue implementation. Valid values: 1U, 2U, or 4U; default 1U
74//!
75//! @description
76//! This macro can be defined in the QF ports to configure the QP::QEQueueCtr
77//! type. If the macro is not defined, the default of 1 byte will be chosen
78//! in qequeue.hpp. The valid #QF_EQUEUE_CTR_SIZE values of 1U, 2U, or 4U,
79//! correspond to QP::QEQueueCtr of uint8_t, uint16_t, and uint32_t,
80//! respectively. The QP::QEQueueCtr data type determines the dynamic range
81//! of numerical values of ring-buffer counters inside event queues, or,
82//! in other words, the maximum number of events that the native QF event
83//! queue can manage.
84//! @sa QP::QEQueue
85#define QF_EQUEUE_CTR_SIZE 1U
86
87//! The size (in bytes) of the block-size representation in the
88//! native QF event pool. Valid values: 1U, 2U, or 4U; default 2U.
89//! #QF_EVENT_SIZ_SIZE.
90//!
91//! @description
92//! This macro can be defined in the QF ports to configure the QP::QMPoolSize
93//! type. If the macro is not defined, the default of #QF_EVENT_SIZ_SIZE
94//! will be chosen in qmpool.hpp, because the memory pool is primarily used for
95//! implementing event pools.
96//!
97//! The valid #QF_MPOOL_SIZ_SIZE values of 1U, 2U, or 4U, correspond to
98//! QP::QMPoolSize of uint8_t, uint16_t, and uint32_t, respectively. The
99//! QP::QMPoolSize data type determines the dynamic range of block-sizes that
100//! the native QP::QMPool can hanle.
101//! @sa #QF_EVENT_SIZ_SIZE, QP::QMPool
102#define QF_MPOOL_SIZ_SIZE 2U

160//! Define the type of the critical section status.
161//!
162//! @description
163//! Defining this macro configures the "saving and restoring critical section
164//! status" policy. Conversely, if this macro is not defined, the simple
165//! "unconditional critical section exit" is used.
166//!
167#define QF_CRIT_STAT_TYPE unsigned
168
169//! Define the critical section entry policy.
170//!
171//! This macro enters a critical section (often by means of disabling
172//! interrupts). When the "saving and restoring critical section status"
173//! policy is used, the macro sets the \a status_ argument to the critical
174//! section status just before the entry. When the policy of "unconditional
175//! critical section exit" is used, the macro does not use the \a status_
176//! argument.
177//!
178//! @note
179//! The QF_CRIT_ENTRY() macro should always be used in pair with the
180//! macro QF_CRIT_EXIT().
181//!
182#define QF_CRIT_ENTRY(stat_) ((stat_) = critEntry())
183
184//! Define the critical section exit policy.
185//!
186//! @description
187//! This macro enters a critical section (often by means of disabling
188//! interrupts). When the "saving and restoring critical section status"
189//! policy is used, the macro restores the critical section status from the
190//! @a status_ argument. When the policy of "unconditional critical section
191//! exit" is used, the macro does not use the \a status argument and
192//! exits the critical section unconditionally (often by means of enabling
193//! interrupts).
194//!
195//! @note
196//! The QF_CRIT_ENTRY() macro should always be used in pair with the
197//! macro QF_CRIT_EXIT().
198//!
199#define QF_CRIT_EXIT(stat_) critExit(stat_)

234//! Platform-dependent macro defining how QF should block the calling
235//! task when the QF native queue is empty
236//!
237//! @note
238//! This is just an example of QACTIVE_EQUEUE_WAIT_(). QK never activates
239//! a task that has no events to process, so in this case the macro asserts
240//! that the queue is not empty. In other QF ports you need to define
241//! the macro appropriately for the underlying kernel/OS you're using.
242#define QACTIVE_EQUEUE_WAIT_(me_) \
243 Q_ASSERT((me_)->m_eQueue.m_frontEvt != nullptr)
244
245//! Platform-dependent macro defining how QF should signal the
246//! active object task that an event has just arrived.
247//!
248//! @description
249//! The macro is necessary only when the native QF event queue is used.
250//! The signaling of task involves unblocking the task if it is blocked.
251//!
252//! @note
253//! QACTIVE_EQUEUE_SIGNAL_() is called from a critical section.
254//! It might leave the critical section internally, but must restore
255//! the critical section before exiting to the caller.
256//!
257//! @note
258//! This is just an example of QACTIVE_EQUEUE_SIGNAL_(). In other QF ports
259//! you need to define the macro appropriately for the underlying
260//! kernel/OS you're using.
261#define QACTIVE_EQUEUE_SIGNAL_(me_) do { \
262 QF::readySet_.insert((me_)->m_prio); \
263 if (QF::intNest_ == 0U) { \
264 uint8_t p = QK_schedPrio_(); \
265 if (p != 0U) { \
266 QK_sched_(p); \
267 } \
268 } \
269} while (false)

289//! Platform-dependent macro defining how QF should obtain an event
290//! @a e_ from the event pool @a p_
291//!
292//! @note
293//! This is a specific implementation for the QK-port of QF.
294//! In other QF ports you need to define the macro appropriately for
295//! the underlying kernel/OS you're using.
296#define QF_EPOOL_GET_(p_, e_, m_, qs_id_) \
297 ((e_) = static_cast<QEvt *>((p_).get((m_), (qs_id_))))
