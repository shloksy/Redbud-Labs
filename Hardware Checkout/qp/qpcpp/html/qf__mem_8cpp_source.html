82void QMPool::init(
83 void * const poolSto,
84 std::uint_fast32_t poolSize,
85 std::uint_fast16_t blockSize) noexcept
86{
87 //! @pre The memory block must be valid and
88 //! the poolSize must fit at least one free block and
89 //! the blockSize must not be too close to the top of the dynamic range
90 Q_REQUIRE_ID(100, (poolSto != nullptr)
91 && (poolSize >= static_cast<std::uint_fast32_t>(sizeof(QFreeBlock)))
92 && (static_cast<std::uint_fast16_t>(blockSize + sizeof(QFreeBlock))
93 > blockSize));
94
95 m_free_head = poolSto;
96
97 // round up the blockSize to fit an integer number of pointers...
98 //start with one
99 m_blockSize = static_cast<QMPoolSize>(sizeof(QFreeBlock));
100
101 //# free blocks in a memory block
102 std::uint_fast16_t nblocks = 1U;
103 while (m_blockSize < static_cast<QMPoolSize>(blockSize)) {
104 m_blockSize += static_cast<QMPoolSize>(sizeof(QFreeBlock));
105 ++nblocks;
106 }
107 // use rounded-up value
108 blockSize = static_cast<std::uint_fast16_t>(m_blockSize);
109
110 // the whole pool buffer must fit at least one rounded-up block
111 Q_ASSERT_ID(110, poolSize >= blockSize);
112
113 // chain all blocks together in a free-list...
114
115 // don't count the last block
116 poolSize -= static_cast<std::uint_fast32_t>(blockSize);
117 m_nTot = 1U; // one (the last) block in the pool
118
119 // start at the head of the free list
120 QFreeBlock *fb = static_cast<QFreeBlock *>(m_free_head);
121
122 // chain all blocks together in a free-list...
123 while (poolSize >= blockSize) {
124 fb->m_next = &fb[nblocks]; // setup the next link
125 fb = fb->m_next; // advance to next block
126 // reduce the available pool size
127 poolSize -= static_cast<std::uint_fast32_t>(blockSize);
128 ++m_nTot; // increment the number of blocks so far
129 }
130
131 fb->m_next = nullptr; // the last link points to NULL
132 m_nFree = m_nTot; // all blocks are free
133 m_nMin = m_nTot; // the minimum number of free blocks
134 m_start = poolSto; // the original start this pool buffer
135 m_end = fb; // the last block in this pool
136}

146 QF_CRIT_E_();
147
148 // have the than margin?
149 QFreeBlock *fb;
150 if (m_nFree > static_cast<QMPoolCtr>(margin)) {
151 fb = static_cast<QFreeBlock *>(m_free_head); // get a free block
152
153 // the pool has some free blocks, so a free block must be available
154 Q_ASSERT_CRIT_(310, fb != nullptr);
155
156 // put volatile to a temporary to avoid UB
157 void * const fb_next = fb->m_next;
158
159 // is the pool becoming empty?
160 m_nFree = (m_nFree - 1U); // one free block less
161 if (m_nFree == 0U) {
162 // pool is becoming empty, so the next free block must be NULL
163 Q_ASSERT_CRIT_(320, fb_next == nullptr);
164
165 m_nMin = 0U;// remember that pool got empty
166 }
167 else {
168 //! @invariant
169 //! The pool is not empty, so the next free-block pointer,
170 //! so the next free block must be in range.

172 //! @tr{PQP18_3}
173
174 // NOTE: The next free block pointer can fall out of range
175 // when the client code writes past the memory block, thus
176 // corrupting the next block.
177 Q_ASSERT_CRIT_(330, QF_PTR_RANGE_(fb_next, m_start, m_end));
178
179 // is the number of free blocks the new minimum so far?
180 if (m_nMin > m_nFree) {
181 m_nMin = m_nFree; // remember the minimum so far

183 }
184

195 else {

198 QS_BEGIN_NOCRIT_PRE_(QS_QF_MPOOL_GET_ATTEMPT, qs_id)
199 QS_TIME_PRE_(); // timestamp
200 QS_OBJ_PRE_(m_start); // the memory managed by this pool

202 QS_MPC_PRE_(margin); // the requested margin
