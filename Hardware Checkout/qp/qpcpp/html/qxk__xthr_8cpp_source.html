308//${QXK::QXThread::post_} ....................................................
309bool QXThread::post_(
310 QEvt const * const e,
311 std::uint_fast16_t const margin,
312 void const * const sender) noexcept
313{
314 QF_CRIT_STAT_
315 QS_TEST_PROBE_DEF(&QXThread::post_)
316
317 // is it the private time event?
318 bool status;
319 if (e == &m_timeEvt) {
320 QF_CRIT_E_();
321 // the private time event is disarmed and not in any queue,
322 // so it is safe to change its signal. The signal of 0 means
323 // that the time event has expired.
324 m_timeEvt.sig = 0U;
325
326 unblock_();
327 QF_CRIT_X_();
328
329 status = true;
330 }
331 // is the event queue provided?
332 else if (m_eQueue.m_end != 0U) {
333
334 //! @pre event pointer must be valid
335 Q_REQUIRE_ID(300, e != nullptr);
336
337 QF_CRIT_E_();
338 QEQueueCtr nFree = m_eQueue.m_nFree; // get volatile into temporary
339
340 // test-probe#1 for faking queue overflow
341 QS_TEST_PROBE_ID(1,
342 nFree = 0U;
343 )
344
345 if (margin == QF::NO_MARGIN) {
346 if (nFree > 0U) {
347 status = true; // can post
348 }
349 else {
350 status = false; // cannot post
351 Q_ERROR_CRIT_(310); // must be able to post the event
352 }
353 }
354 else if (nFree > static_cast<QEQueueCtr>(margin)) {
355 status = true; // can post
356 }
357 else {
358 status = false; // cannot post, but don't assert
359 }
360
361 // is it a dynamic event?
362 if (e->poolId_ != 0U) {
363 QEvt_refCtr_inc_(e); // increment the reference counter
364 }
365
366 if (status) { // can post the event?
367
368 --nFree; // one free entry just used up
369 m_eQueue.m_nFree = nFree; // update the volatile
370 if (m_eQueue.m_nMin > nFree) {
371 m_eQueue.m_nMin = nFree; // update minimum so far
372 }
373
374 QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_POST, m_prio)
375 QS_TIME_PRE_(); // timestamp
376 QS_OBJ_PRE_(sender); // the sender object
377 QS_SIG_PRE_(e->sig); // the signal of the event
378 QS_OBJ_PRE_(this); // this active object
379 QS_2U8_PRE_(e->poolId_, e->refCtr_); // poolID & refCtr
380 QS_EQC_PRE_(nFree); // number of free entries
381 QS_EQC_PRE_(m_eQueue.m_nMin); // min number of free entries
382 QS_END_NOCRIT_PRE_()
383
384 // queue empty?
385 if (m_eQueue.m_frontEvt == nullptr) {
386 m_eQueue.m_frontEvt = e; // deliver event directly
387
388 // is this thread blocked on the queue?
389 if (m_temp.obj == QXK_PTR_CAST_(QMState*, &m_eQueue)) {
390 static_cast<void>(teDisarm_());
391 QF::readySet_.insert(
392 static_cast<std::uint_fast8_t>(m_prio));
393 if (!QXK_ISR_CONTEXT_()) {
394 static_cast<void>(QXK_sched_());
395 }
396 }
397 }
398 // queue is not empty, insert event into the ring-buffer
399 else {
400 // insert event into the ring buffer (FIFO)
401 m_eQueue.m_ring[m_eQueue.m_head] = e;
402
403 // need to wrap the head couner?
404 if (m_eQueue.m_head == 0U) {
405 m_eQueue.m_head = m_eQueue.m_end; // wrap around
406 }
407 // advance the head (counter clockwise)
408 m_eQueue.m_head = (m_eQueue.m_head - 1U);
409 }
410
411 QF_CRIT_X_();
412 }
413 else { // cannot post the event
414
415 QS_BEGIN_NOCRIT_PRE_(QS_QF_ACTIVE_POST_ATTEMPT, m_prio)
416 QS_TIME_PRE_(); // timestamp
417 QS_OBJ_PRE_(sender); // the sender object
418 QS_SIG_PRE_(e->sig); // the signal of the event
419 QS_OBJ_PRE_(this); // this active object (recipient)
420 QS_2U8_PRE_(e->poolId_, e->refCtr_); // poolID & ref Count
421 QS_EQC_PRE_(nFree); // number of free entries
422 QS_EQC_PRE_(margin); // margin
423 QS_END_NOCRIT_PRE_()
424
425 QF_CRIT_X_();
426
427 #if (QF_MAX_EPOOL > 0U)
428 QF::gc(e); // recycle the event to avoid a leak
429 #endif
430 }
431 }
432 else { // the queue is not available
433 #if (QF_MAX_EPOOL > 0U)
434 QF::gc(e); // make sure the event is not leaked
435 #endif
436 status = false;
437 Q_ERROR_ID(320); // this extended thread cannot accept events
438 }
439
440 return status;
441}

468//${QXK::QXThread::teArm_} ...................................................
469void QXThread::teArm_(
470 enum_t const sig,
471 std::uint_fast16_t const nTicks) noexcept
472{
473 //! @pre the time event must be unused
474 Q_REQUIRE_ID(700, m_timeEvt.m_ctr == 0U);
475
476 m_timeEvt.sig = static_cast<QSignal>(sig);
477
478 if (nTicks != QXTHREAD_NO_TIMEOUT) {
479 m_timeEvt.m_ctr = static_cast<QTimeEvtCtr>(nTicks);
480 m_timeEvt.m_interval = 0U;
481
482 // is the time event unlinked?
483 // NOTE: For the duration of a single clock tick of the specified tick
484 // rate a time event can be disarmed and yet still linked in the list,
485 // because un-linking is performed exclusively in QTimeEvt::tickX().
486 if (static_cast<std::uint8_t>(m_timeEvt.refCtr_ & TE_IS_LINKED) == 0U)
487 {
488 std::uint_fast8_t const tickRate =
489 static_cast<std::uint_fast8_t>(m_timeEvt.refCtr_);
490
491 // mark as linked
492 m_timeEvt.refCtr_ = static_cast<std::uint8_t>(
493 m_timeEvt.refCtr_ | TE_IS_LINKED);
494
495 // The time event is initially inserted into the separate
496 // "freshly armed" list based on timeEvtHead_[tickRate].act.
497 // Only later, inside QTimeEvt::tick_(), the "freshly armed"
498 // list is appended to the main list of armed time events based on
499 // timeEvtHead_[tickRate].next. Again, this is to keep any
500 // changes to the main list exclusively inside QTimeEvt::tick_().
501 //
502 m_timeEvt.m_next
503 = QXK_PTR_CAST_(QTimeEvt*,
504 QTimeEvt::timeEvtHead_[tickRate].m_act);
505 QTimeEvt::timeEvtHead_[tickRate].m_act = &m_timeEvt;
506 }
507 }
