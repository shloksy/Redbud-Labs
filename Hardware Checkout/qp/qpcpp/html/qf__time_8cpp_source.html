278//${QF::QTimeEvt::tick_} .....................................................
279void QTimeEvt::tick_(
280 std::uint_fast8_t const tickRate,
281 void const * const sender) noexcept
282{
283 Q_UNUSED_PAR(sender); // when Q_SPY not defined
284
285 QTimeEvt *prev = &timeEvtHead_[tickRate];
286
287 QF_CRIT_STAT_
288 QF_CRIT_E_();
289
290 QS_BEGIN_NOCRIT_PRE_(QS_QF_TICK, 0U)
291 prev->m_ctr = (prev->m_ctr + 1U);
292 QS_TEC_PRE_(prev->m_ctr); // tick ctr
293 QS_U8_PRE_(tickRate); // tick rate
294 QS_END_NOCRIT_PRE_()
295
296 // scan the linked-list of time events at this rate...
297 for (;;) {
298 QTimeEvt *t = prev->m_next; // advance down the time evt. list
299
300 // end of the list?
301 if (t == nullptr) {
302
303 // any new time events armed since the last run of tick_()?
304 if (timeEvtHead_[tickRate].m_act != nullptr) {
305
306 // sanity check
307 Q_ASSERT_CRIT_(110, prev != nullptr);
308 prev->m_next = timeEvtHead_[tickRate].toTimeEvt();
309 timeEvtHead_[tickRate].m_act = nullptr;
310 t = prev->m_next; // switch to the new list
311 }
312 else {
313 break; // all currently armed time evts. processed
314 }
315 }
316
317 // time event scheduled for removal?
318 if (t->m_ctr == 0U) {
319 prev->m_next = t->m_next;
320 // mark time event 't' as NOT linked
321 t->refCtr_ = static_cast<std::uint8_t>(t->refCtr_
322 & static_cast<std::uint8_t>(~TE_IS_LINKED));
323 // do NOT advance the prev pointer
324 QF_CRIT_X_(); // exit crit. section to reduce latency
325
326 // prevent merging critical sections, see NOTE1 below
327 QF_CRIT_EXIT_NOP();
328 }
329 else {
330 t->m_ctr = (t->m_ctr - 1U);
331
332 // is time evt about to expire?
333 if (t->m_ctr == 0U) {
334 QActive * const act = t->toActive(); // temp for volatile
335
336 // periodic time evt?
337 if (t->m_interval != 0U) {
338 t->m_ctr = t->m_interval; // rearm the time event
339 prev = t; // advance to this time event
340 }
341 // one-shot time event: automatically disarm
342 else {
343 prev->m_next = t->m_next;
344
345 // mark time event 't' as NOT linked
346 t->refCtr_ = static_cast<std::uint8_t>(t->refCtr_
347 & static_cast<std::uint8_t>(~TE_IS_LINKED));
348 // do NOT advance the prev pointer
349
350 QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_AUTO_DISARM,
351 act->m_prio)
352 QS_OBJ_PRE_(t); // this time event object
353 QS_OBJ_PRE_(act); // the target AO
354 QS_U8_PRE_(tickRate); // tick rate
355 QS_END_NOCRIT_PRE_()
356 }
357
358 QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_POST, act->m_prio)
359 QS_TIME_PRE_(); // timestamp
360 QS_OBJ_PRE_(t); // the time event object
361 QS_SIG_PRE_(t->sig); // signal of this time event
362 QS_OBJ_PRE_(act); // the target AO
363 QS_U8_PRE_(tickRate); // tick rate
364 QS_END_NOCRIT_PRE_()
365
366 QF_CRIT_X_(); // exit crit. section before posting
367
368 // asserts if queue overflows
369 act->POST(t, sender);
370 }
371 else {
372 prev = t; // advance to this time event
373 QF_CRIT_X_(); // exit crit. section to reduce latency
374
375 // prevent merging critical sections
376 // In some QF ports the critical section exit takes effect only
377 // on the next machine instruction. If this case, the next
378 // instruction is another entry to a critical section, the
379 // critical section won't be really exited, but rather the
380 // two adjacent critical sections would be merged. The
381 // QF_CRIT_EXIT_NOP() macro contains minimal code required
382 // to prevent such merging of critical sections in QF ports,
383 // in which it can occur.
384 QF_CRIT_EXIT_NOP();
385 }
386 }
387 QF_CRIT_E_(); // re-enter crit. section to continue
388 }
389 QF_CRIT_X_();
390}
