259void QTimeEvt::tick1_(
260 std::uint_fast8_t const tickRate,

262{
263 // The testing version of system tick processing performs as follows:
264 // 1. If the Current Time Event (TE) Object is defined and the TE is armed,
265 // the TE is disarmed (if one-shot) and then posted to the recipient AO.
266 // 2. The linked-list of all armed Time Events is updated.
267 //
268
269 QF_CRIT_STAT_
270 QF_CRIT_E_();
271
272 QTimeEvt *prev = &QTimeEvt::timeEvtHead_[tickRate];
273
274 QS_BEGIN_NOCRIT_PRE_(QS_QF_TICK, 0U)
275 prev->m_ctr = (prev->m_ctr + 1U);
276 QS_TEC_PRE_(prev->m_ctr); // tick ctr
277 QS_U8_PRE_(tickRate); // tick rate
278 QS_END_NOCRIT_PRE_()
279
280 // is current Time Event object provided?
281 QTimeEvt *t = static_cast<QTimeEvt *>(QS::rxPriv_.currObj[QS::TE_OBJ]);
282 if (t != nullptr) {
283
284 // the time event must be armed
285 Q_ASSERT_ID(810, t->m_ctr != 0U);
286
287 // temp. for volatile
288 QActive * const act = static_cast<QActive *>(t->m_act);
289
290 // the recipient AO must be provided
291 Q_ASSERT_ID(820, act != nullptr);
292
293 // periodic time evt?
294 if (t->m_interval != 0U) {
295 t->m_ctr = t->m_interval; // rearm the time event
296 }
297 else { // one-shot time event: automatically disarm
298 t->m_ctr = 0U; // auto-disarm
299 // mark time event 't' as NOT linked
300 t->refCtr_ = static_cast<std::uint8_t>(t->refCtr_
301 & static_cast<std::uint8_t>(~TE_IS_LINKED));
302
303 QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_AUTO_DISARM, act->m_prio)
304 QS_OBJ_PRE_(t); // this time event object
305 QS_OBJ_PRE_(act); // the target AO
306 QS_U8_PRE_(tickRate); // tick rate
307 QS_END_NOCRIT_PRE_()
308 }
309
310 QS_BEGIN_NOCRIT_PRE_(QS_QF_TIMEEVT_POST, act->m_prio)
311 QS_TIME_PRE_(); // timestamp
312 QS_OBJ_PRE_(t); // the time event object
313 QS_SIG_PRE_(t->sig); // signal of this time event
314 QS_OBJ_PRE_(act); // the target AO
315 QS_U8_PRE_(tickRate); // tick rate
316 QS_END_NOCRIT_PRE_()
317
318 QF_CRIT_X_(); // exit crit. section before posting
319
320 // asserts if queue overflows
321 static_cast<void>(act->POST(t, sender));
322
323 QF_CRIT_E_();
324 }
325
326 // update the linked list of time events
327 for (;;) {
328 t = prev->m_next; // advance down the time evt. list
329
330 // end of the list?
331 if (t == nullptr) {
332
333 // any new time events armed since the last run of tick_()?
334 if (QTimeEvt::timeEvtHead_[tickRate].m_act != nullptr) {

341 }
342 else {
343 break; // all currently armed time evts. processed
344 }
345 }
346
347 // time event scheduled for removal?
348 if (t->m_ctr == 0U) {
349 prev->m_next = t->m_next;
350 // mark time event 't' as NOT linked
351 t->refCtr_ = static_cast<std::uint8_t>(t->refCtr_

353 // do NOT advance the prev pointer

355
356 // prevent merging critical sections, see NOTE1 below
357 QF_CRIT_EXIT_NOP();

361 QF_CRIT_X_(); // exit crit. section to reduce latency
362
363 // prevent merging critical sections, see NOTE1 below
364 QF_CRIT_EXIT_NOP();

367 }
