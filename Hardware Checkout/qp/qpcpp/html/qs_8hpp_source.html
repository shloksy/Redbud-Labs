829//${QS::QS-rx::rxPut} ........................................................
830//! Put one byte into the QS RX lock-free buffer
831inline bool rxPut(std::uint8_t const b) noexcept {
832 QSCtr head = rxPriv_.head + 1U;
833 if (head == rxPriv_.end) {
834 head = 0U;
835 }
836 if (head != rxPriv_.tail) { // buffer NOT full?
837 rxPriv_.buf[rxPriv_.head] = b;
838 rxPriv_.head = head;
839 return true; // byte placed in the buffer
840 }
841 else {
842 return false; // byte NOT placed in the buffer
843 }
844}
