278void QK_activate_(void) {
279 uint8_t const prio_in = QK_attr_.actPrio; /* saved initial priority */
280 uint8_t p = QK_attr_.nextPrio; /* next prio to run */
281 QK_attr_.nextPrio = 0U; /* clear for the next time */
282
283 /* QK_attr_.actPrio and QK_attr_.nextPrio must be in range */
284 Q_REQUIRE_ID(500, (prio_in <= QF_MAX_ACTIVE)
285 && (0U < p) && (p <= QF_MAX_ACTIVE));
286
287 #if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
288 uint8_t pprev = prio_in;
289 #endif /* QF_ON_CONTEXT_SW || Q_SPY */
290
291 /* loop until no more ready-to-run AOs of higher pthre than the initial */
292 QActive *a;
293 do {
294 a = QActive_registry_[p]; /* obtain the pointer to the AO */
295 Q_ASSERT_ID(505, a != (QActive *)0); /* the AO must be registered */
296
297 /* set new active priority and preemption-threshold */
298 QK_attr_.actPrio = p;
299 QK_attr_.actThre = a->pthre;
300
301 #if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
302 if (p != pprev) { /* changing threads? */
303
304 QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, p)
305 QS_TIME_PRE_(); /* timestamp */
306 QS_2U8_PRE_(p, /* priority of the scheduled AO */
307 pprev); /* previous priority */
308 QS_END_NOCRIT_PRE_()
309
310 #ifdef QF_ON_CONTEXT_SW
311 QF_onContextSw(((pprev != 0U)
312 ? QActive_registry_[pprev]
313 : (QActive *)0), a);
314 #endif /* QF_ON_CONTEXT_SW */
315
316 pprev = p; /* update previous priority */
317 }
318 #endif /* QF_ON_CONTEXT_SW || Q_SPY */
319
320 QF_INT_ENABLE(); /* unconditionally enable interrupts */
321
322 /* perform the run-to-completion (RTC) step...
323 * 1. retrieve the event from the AO's event queue, which by this
324 * time must be non-empty and QActive_get_() asserts it.
325 * 2. dispatch the event to the AO's state machine.
326 * 3. determine if event is garbage and collect it if so
327 */
328 QEvt const * const e = QActive_get_(a);
329 QHSM_DISPATCH(&a->super, e, p);
330 #if (QF_MAX_EPOOL > 0U)
331 QF_gc(e);
332 #endif
333
334 /* determine the next highest-priority AO ready to run... */
335 QF_INT_DISABLE(); /* unconditionally disable interrupts */
336
337 if (a->eQueue.frontEvt == (QEvt *)0) { /* empty queue? */
338 QPSet_remove(&QF_readySet_, p);
339 }
340
341 if (QPSet_isEmpty(&QF_readySet_)) {
342 p = 0U; /* no activation needed */
343 }
344 else {
345 /* find new highest-prio AO ready to run... */
346 p = (uint8_t)QPSet_findMax(&QF_readySet_);
347
348 /* is the new priority below the initial preemption-threshold? */
349 if (p <= QActive_registry_[prio_in]->pthre) {
350 p = 0U; /* no activation needed */
351 }
352 /* is the AO's priority below the lock preemption-ceiling? */
353 else if (p <= QK_attr_.lockCeil) {
354 p = 0U; /* no activation needed */
355 }
356 else {
357 Q_ASSERT_ID(510, p <= QF_MAX_ACTIVE);
358 }
359 }
360 } while (p != 0U);
361
362 /* restore the active priority and preemption-threshold */
363 QK_attr_.actPrio = prio_in;
364 QK_attr_.actThre = QActive_registry_[prio_in]->pthre;
365
366 #if (defined QF_ON_CONTEXT_SW) || (defined Q_SPY)
367 if (prio_in != 0U) { /* resuming an active object? */
368 a = QActive_registry_[prio_in]; /* pointer to the preempted AO */
369
370 QS_BEGIN_NOCRIT_PRE_(QS_SCHED_NEXT, prio_in)
371 QS_TIME_PRE_(); /* timestamp */
372 /* priority of the resumed AO, previous priority */
373 QS_2U8_PRE_(prio_in, pprev);
374 QS_END_NOCRIT_PRE_()
375 }
376 else { /* resuming priority==0 --> idle */
377 a = (QActive *)0; /* QK idle loop */
378
379 QS_BEGIN_NOCRIT_PRE_(QS_SCHED_IDLE, pprev)
380 QS_TIME_PRE_(); /* timestamp */
381 QS_U8_PRE_(pprev); /* previous priority */
382 QS_END_NOCRIT_PRE_()
383 }
384

387 #endif /* QF_ON_CONTEXT_SW */
388
389 #endif /* QF_ON_CONTEXT_SW || Q_SPY */
