433/*${QEP::QHsm::tran_} ......................................................*/
434int_fast8_t QHsm_tran_(QHsm * const me,
435 QStateHandler * const path,
436 uint_fast8_t const qs_id)
437{
438 #ifndef Q_SPY
439 Q_UNUSED_PAR(qs_id);
440 #endif
441
442 int_fast8_t ip = -1; /* transition entry path index */
443 QStateHandler t = path[0];
444 QStateHandler const s = path[2];
445
446 /* (a) check source==target (transition to self)... */
447 if (s == t) {
448 (void)QHsm_state_exit_(me, s, qs_id); /* exit source */
449 ip = 0; /* enter the target */
450 }
451 else {
452 /* find superstate of target */
453 (void)QHsm_reservedEvt_(me, t, Q_EMPTY_SIG);
454
455 t = me->temp.fun;
456
457 /* (b) check source==target->super... */
458 if (s == t) {
459 ip = 0; /* enter the target */
460 }
461 else {
462 /* find superstate of src */
463 (void)QHsm_reservedEvt_(me, s, Q_EMPTY_SIG);
464
465 /* (c) check source->super==target->super... */
466 if (me->temp.fun == t) {
467 (void)QHsm_state_exit_(me, s, qs_id); /* exit source */
468 ip = 0; /* enter the target */
469 }
470 else {
471 /* (d) check source->super==target... */
472 if (me->temp.fun == path[0]) {
473 (void)QHsm_state_exit_(me, s, qs_id); /* exit source */
474 }
475 else {
476 /* (e) check rest of source==target->super->super..
477 * and store the entry path along the way
478 */
479 int_fast8_t iq = 0; /* indicate that LCA not found */
480 ip = 1; /* enter target and its superstate */
481 path[1] = t; /* save the superstate of target */
482 t = me->temp.fun; /* save source->super */
483
484 /* find target->super->super... */
485 QState r = QHsm_reservedEvt_(me, path[1], Q_EMPTY_SIG);
486 while (r == Q_RET_SUPER) {
487 ++ip;
488 path[ip] = me->temp.fun; /* store the entry path */
489 if (me->temp.fun == s) { /* is it the source? */
490 iq = 1; /* indicate that LCA found */
491
492 /* entry path must not overflow */
493 Q_ASSERT_ID(510,
494 ip < QHSM_MAX_NEST_DEPTH_);
495 --ip; /* do not enter the source */
496 r = Q_RET_HANDLED; /* terminate loop */
497 }
498 /* it is not the source, keep going up */
499 else {
500 r = QHsm_reservedEvt_(me, me->temp.fun,
501 Q_EMPTY_SIG);
502 }
503 }
504
505 /* the LCA not found yet? */
506 if (iq == 0) {
507
508 /* entry path must not overflow */
509 Q_ASSERT_ID(520, ip < QHSM_MAX_NEST_DEPTH_);
510
511 /* exit source */
512 (void)QHsm_state_exit_(me, s, qs_id);
513
514 /* (f) check the rest of source->super
515 * == target->super->super...
516 */
517 iq = ip;
518 r = Q_RET_IGNORED; /* LCA NOT found */
519 do {
520 if (t == path[iq]) { /* is this the LCA? */
521 r = Q_RET_HANDLED; /* LCA found */
522 ip = iq - 1; /* do not enter LCA */
523 iq = -1; /* cause termintion of the loop */
524 }
525 else {
526 --iq; /* try lower superstate of target */
527 }
528 } while (iq >= 0);
529
530 /* LCA not found? */
531 if (r != Q_RET_HANDLED) {
532 /* (g) check each source->super->...
533 * for each target->super...
534 */
535 r = Q_RET_IGNORED; /* keep looping */
536 do {
537 /* exit from t handled? */
538 if (QHsm_state_exit_(me, t, qs_id)) {
539 /* find superstate of t */
540 (void)QHsm_reservedEvt_(me, t, Q_EMPTY_SIG);
541 }
542 t = me->temp.fun; /* set to super of t */
543 iq = ip;
544 do {
545 /* is this LCA? */
546 if (t == path[iq]) {
547 /* do not enter LCA */
548 ip = (int_fast8_t)(iq - 1);
549 iq = -1; /* break out of inner loop */
550 /* break out of outer loop */
551 r = Q_RET_HANDLED;
552 }
553 else {
554 --iq;
555 }
556 } while (iq >= 0);
557 } while (r != Q_RET_HANDLED);
558 }
559 }
560 }
561 }
562 }
563 }
564 return ip;
